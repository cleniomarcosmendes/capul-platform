    -- Dados do Produto (SB1)
    SB1.B1_COD AS CODIGO_PRODUTO,
    SB1.B1_DESC AS DESCRICAO_PRODUTO,
    SB1.B1_UPRC AS ULTIMO_PRECO_COMPRA_SB1,
    
    -- Dados Complementares do Produto (SBZ)

    -- Dados de Estoque (SB2)
    SB2.B2_LOCAL AS ARMAZEM,
    SB2.B2_QATU AS QUANTIDADE_ATUAL,
    SB2.B2_VATU1 AS VALOR_ATUAL,
    SB2.B2_CM1 AS CUSTO_MEDIO,

    LEFT JOIN SBZ010 SBZ ON (
        SBZ.BZ_FILIAL = '01' 
        AND SBZ.BZ_COD = SB1.B1_COD
        AND SBZ.D_E_L_E_T_ = ' '

        SB2.B2_FILIAL = '01'
        AND SB2.B2_COD = SB1.B1_COD
        AND SB2.D_E_L_E_T_ = ' '

        AND DA1.DA1_CODTAB = 'AGV'
        AND DA1.DA1_CODPRO = SB1.B1_COD
        AND DA1.D_E_L_E_T_ = ' '
